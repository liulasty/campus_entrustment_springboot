-- 创建通知表记录用户是否已读通知
CREATE TABLE NotificationReadStatus(
    -- 唯一标识每条通知记录的ID
    NotificationID INT NOT NULL COMMIT '通知ID',
    
    -- 与通知关联的用户ID
	UserID INT NOT NULL COMMIT '通知用户ID',

    -- 用户阅读通知的时间，若未读则为NULL
    ReadTime DATETIME DEFAULT NULL COMMIT '用户阅读通知的时间',

    -- 是否已读标志，0表示未读，1表示已读
    IsRead TINYINT(1) NOT NULL DEFAULT 0 COMMIT '是否已读标志'
)COMMIT='通知表记录用户是否已读通知';
